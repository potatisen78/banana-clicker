import pygame
from sys import exit
pygame.init()




    #fsf = pygame.font.Font("Pixeltype.ttf", 50)
    #scoresurf = fsf.render(gametime, True, tcoul)


screen = pygame.display.set_mode((800,400))
pygame.display.set_caption("crying but not logikaly frying")
fps = (pygame.time.Clock())
playergravity = 0
gameactive = False
starttime = 0
tcoul = (64, 64, 64)
def displayscore():
    gametime = int(pygame.time.get_ticks() / 1000) - starttime
    fsf = pygame.font.Font("Pixeltype.ttf", 50)
    scoresurf = fsf.render(str(gametime), False, tcoul)
    scorerect = scoresurf.get_rect(center = (400, 50))
    screen.blit(scoresurf, scorerect)




surface1 = pygame.image.load("sky.png").convert()
ground = pygame.image.load("ground.png").convert()
#scorets = fsf.render("hello there", False, tcoul )
#scorerect = scorets.get_rect(center = (400, 50))

snails = pygame.image.load("snail1.png").convert_alpha()
snailrect = snails.get_rect(midbottom= (600, 300 ))



player = pygame.image.load("player_walk_1.png").convert_alpha()
playerrect = player.get_rect(midbottom = (80, 300))

while True:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
            exit()

        if event.type == pygame.KEYDOWN:
            if event.key == pygame.K_SPACE:
                if playerrect.bottom >= 300:
                            playergravity = -20

        if gameactive == False:
            if event.type == pygame.KEYDOWN:
                if event.key == pygame.K_SPACE:
                    gameactive = True
                    starttime = int(pygame.time.get_ticks() / 1000)




    if event.type == pygame.MOUSEBUTTONDOWN:
        if playerrect.collidepoint(event.pos):
            if playerrect.bottom >= 300:
                playergravity = -20



    if gameactive:
        screen.blit(surface1, (0, 0))
        screen.blit(ground, (0, 300))
        #screen.blit(scorets, scorerect)
        displayscore()

        # PLAYER
        if playerrect.bottom < 300:
            playergravity += 1
        if playerrect.bottom >= 300:
            playerrect.bottom = 300
        screen.blit(player, playerrect)
        playerrect.y += playergravity

        # keys = pygame.key.get_pressed()
        # if keys[pygame.K_SPACE]:
        # print("jump")

        if snailrect.left < 0:
            snailrect.left = 800
        snailrect.x -= 4
        screen.blit(snails, snailrect)

        mousepos = pygame.mouse.get_pos()

        #collision
        if snailrect.colliderect(playerrect):
            gameactive = False
            if playerrect.colliderect(snailrect):
                gameactive = False
    else:
        screen.fill((94,129,162))

    pygame.display.update()
    fps.tick(60)
